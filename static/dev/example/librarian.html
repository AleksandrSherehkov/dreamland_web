Библиотекарша в Мидгаардской Библиотеке, реагирует на фразу 'кофе, пожалуйста'.

librarianInit = function () {
    .get_mob_index(28800).onSpeech = function (ch, vict, msg) 
    {
	if(msg.match("кофе|coffe")
	    && msg.match("пожалуй*ста|please"))
	{
	    var cofee;

	    cofee = .get_obj_index(3101).create();
	    ch.interpret("emote снимает с печки чайник.");
	    ch.interpret("emote наливает чашку {Dкофе{x.");
	    cofee.obj_to_char(ch);
	    ch.interpret("give cup " + vict.getName());
	    ch.interpret("emote тепло улыбается.");
	}
    };
}
