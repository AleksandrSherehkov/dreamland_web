Пчелы в Пригороде Соласа периодически залезают в леток и обратно.

ragnarInit = function() {
    // bees flying to-from 'letok'
    var bee;

    bee = .get_mob_index(17600);

    bee.room1_vnum = 17606;
    bee.room2_vnum = 17618;
    bee.wait_time = 15;
    bee.extra_exit = "letok";

    bee.onSpec = function(b){
	if (b.cur_time == null) {
	    b.cur_time = .number_range(0, wait_time);
	    return;
	}
	
	if (b.cur_time >= 0) {
	    b.cur_time = b.cur_time - 1;
	    return;
	}

	if (b.in_room.vnum == room1_vnum || b.in_room.vnum == room2_vnum) {
	    b.interpret("walk " + extra_exit);
	    b.cur_time = wait_time;
	}
	
    };
}
